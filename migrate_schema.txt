\COPY (SELECT 'ALTER TABLE ' || table_schema || '.' || '"' || table_name || '"' || ' SET SCHEMA backend;'
FROM information_schema.tables
WHERE table_schema = 'public' AND table_type = 'BASE TABLE') TO move_schema.txt;

## COPY THE OUTPUT ON move_schema.txt AND RUN IT;

SELECT 'ALTER TYPE ' || n.nspname || '.' || '"' || t.typname || '"' || ' SET SCHEMA backend;'
FROM pg_type t
JOIN pg_enum e ON t.oid = e.enumtypid
JOIN pg_namespace n ON n.oid = t.typnamespace
WHERE n.nspname = 'public'
GROUP BY n.nspname, t.typname;
